```
# 1. 进入本地仓库目录
cd /path/to/your/github/repo

# 2. 确保当前在主分支（如 main），且工作区干净（无未提交修改）
# 查看状态，有未提交文件需先提交或 stash
git status
# 若有未提交文件，可选：git stash（暂存）或 git commit -m "临时提交"

# 3. 创建孤儿分支（命名为 clean_branch，可自定义）
git checkout --orphan clean_branch

# 4. 此时所有文件都是未暂存状态，执行 git add 暂存所有文件
git add --all

# 5. 完成首次提交（这是新分支的第一个提交，也是唯一的历史）
git commit -m "Initial commit (clean history)"

# 6. 删除原有主分支（如 main，注意：删除前确保代码已备份）
git branch -D main

# 7. 将孤儿分支重命名为 main（替换原主分支）
git branch -m clean_branch main

# 8. 强制推送到 GitHub（覆盖远程仓库的所有历史，不可逆！）
git push -f origin main
```
